import uuid

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session
from app.api import deps
from app.models.organization import Organization
from app.models.employee import Employee
from app.models.payroll import EmployeeSalary, SalaryComponent, Arrear, OneTimePayment
from app.schemas.payroll_bulk_operations import (
    BulkSalaryRevisionCreate, BulkComponentAdjustmentCreate, 
    ArrearCreate, OneTimePaymentCreate, BulkOperationResult
)
from app.core.permissions import PayrollSalaryComponentPermissions

router = APIRouter()

def _get_org_id(current_user: Union[Organization, Employee]) -> int:
    return current_user.id if isinstance(current_user, Organization) else current_user.organization_id

def _require_permission(db, current_user, code, action_label):
    if not isinstance(current_user, Organization) and not deps.has_permission(db, current_user, code):
        raise HTTPException(status_code=403, detail=f"Permission denied: {action_label}")

@router.post("/salary-revision", response_model=BulkOperationResult)
def bulk_salary_revision(
    data: BulkSalaryRevisionCreate,
    db: Session = Depends(deps.get_db),
    current_user: Union[Organization, Employee] = Depends(deps.get_current_user)
):
    _require_permission(db, current_user, PayrollSalaryComponentPermissions.UPDATE, "bulk revise salary")
    org_id = _get_org_id(current_user)
    
    employees = db.query(Employee).filter(Employee.organization_id == org_id, Employee.uuid.in_(data.employee_uuids)).all()
    emp_map = {str(e.uuid): e.id for e in employees}
    
    success, failures = 0, []
    for uuid_str in data.employee_uuids:
        emp_id = emp_map.get(uuid_str)
        if not emp_id:
            failures.append({"uuid": uuid_str, "error": "Employee not found"})
            continue
            
        salary = db.query(EmployeeSalary).filter(EmployeeSalary.employee_id == emp_id, EmployeeSalary.is_active == True).first()
        if not salary:
            failures.append({"uuid": uuid_str, "error": "No active salary found"})
            continue
            
        multiplier = (1 + (data.revision_percentage / 100)).quantize(Decimal("0.01"), rounding=ROUND_HALF_UP)
        salary.annual_ctc = (salary.annual_ctc * multiplier).quantize(Decimal("0.01"), rounding=ROUND_HALF_UP)
        salary.monthly_ctc = (salary.annual_ctc / 12).quantize(Decimal("0.01"), rounding=ROUND_HALF_UP)
        db.commit()
        success += 1
        
    return {"success_count": success, "failure_count": len(failures), "errors": failures}

@router.post("/arrears", response_model=BulkOperationResult)
def bulk_create_arrears(
    data: List[ArrearCreate],
    db: Session = Depends(deps.get_db),
    current_user: Union[Organization, Employee] = Depends(deps.get_current_user)
):
    _require_permission(db, current_user, PayrollSalaryComponentPermissions.CREATE, "bulk create arrears")
    org_id = _get_org_id(current_user)
    
    uuids = [d.employee_uuid for d in data]
    employees = db.query(Employee).filter(Employee.organization_id == org_id, Employee.uuid.in_(uuids)).all()
    emp_map = {str(e.uuid): e.id for e in employees}
    
    success, failures = 0, []
    for item in data:
        emp_id = emp_map.get(item.employee_uuid)
        if not emp_id:
            failures.append({"uuid": item.employee_uuid, "error": "Employee not found"})
            continue
        arrear = Arrear(organization_id=org_id, employee_id=emp_id, **item.model_dump(exclude={'employee_uuid'}), arrear_number=str(uuid.uuid4())[:8])
        db.add(arrear)
        success += 1
    db.commit()
    return {"success_count": success, "failure_count": len(failures), "errors": failures}
